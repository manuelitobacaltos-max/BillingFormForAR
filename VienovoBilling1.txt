<script>
  // ─── STATE ───────────────────────────────────────────────────────────────────
  let invoices = JSON.parse(localStorage.getItem('invoices') || '[]');
  let config   = JSON.parse(localStorage.getItem('ejConfig') || '{}');
  let rowCount = 0;

  // ─── INIT ─────────────────────────────────────────────────────────────────────
  window.onload = () => {
    document.getElementById('headerDate').textContent = new Date().toLocaleDateString('en-PH', { weekday:'long', year:'numeric', month:'long', day:'numeric' });
    document.getElementById('invDate').valueAsDate = new Date();
    const due = new Date(); due.setDate(due.getDate() + 30);
    document.getElementById('dueDate').valueAsDate = due;
    document.getElementById('invNumber').value = 'INV-' + String(invoices.length + 1).padStart(4,'0');
    addRow(); addRow();
    loadConfig();
    renderList();
    checkDeadlines();
    if (config.publicKey) emailjs.init(config.publicKey);
  };

  // ─── TABS ─────────────────────────────────────────────────────────────────────
  function switchTab(t) {
    document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
    document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
    event.target.classList.add('active');
    document.getElementById('tab-' + t).classList.add('active');
    if (t === 'list') renderList();
  }

  // ─── ROWS ─────────────────────────────────────────────────────────────────────
  function addRow() {
    const id = rowCount++;
    const tr = document.createElement('tr');
    tr.id = 'row-' + id;
    tr.innerHTML = `
      <td><input type="text" placeholder="Service / Product description" onchange="calcTotals()" /></td>
      <td><input type="number" value="1" min="0" style="text-align:center" onchange="calcTotals()" /></td>
      <td><input type="number" value="0" min="0" step="0.01" onchange="calcTotals()" /></td>
      <td><input type="number" value="0" readonly style="background:#f7f9fc;" /></td>
      <td><button class="del-btn" onclick="delRow(${id})">✕</button></td>`;
    document.getElementById('itemsBody').appendChild(tr);
    calcTotals();
  }

  function delRow(id) {
    const r = document.getElementById('row-' + id);
    if (r) r.remove();
    calcTotals();
  }

  function calcTotals() {
    let sub = 0;
    document.querySelectorAll('#itemsBody tr').forEach(tr => {
      const inputs = tr.querySelectorAll('input');
      const qty = parseFloat(inputs[1].value) || 0;
      const price = parseFloat(inputs[2].value) || 0;
      const amt = qty * price;
      inputs[3].value = amt.toFixed(2);
      sub += amt;
    });
    const tax = sub * 0.12;
    document.getElementById('subtotal').textContent  = sub.toFixed(2);
    document.getElementById('taxAmt').textContent    = tax.toFixed(2);
    document.getElementById('grandTotal').textContent = (sub + tax).toFixed(2);
  }

  // ─── COLLECT FORM ─────────────────────────────────────────────────────────────
  function collectForm() {
    const items = [];
    document.querySelectorAll('#itemsBody tr').forEach(tr => {
      const inputs = tr.querySelectorAll('input');
      const desc = inputs[0].value.trim();
      if (desc) items.push({ desc, qty: inputs[1].value, price: inputs[2].value, amt: inputs[3].value });
    });
    return {
      id: Date.now(),
      invNumber:   document.getElementById('invNumber').value.trim(),
      fromCompany: document.getElementById('fromCompany').value.trim(),
      custName:    document.getElementById('custName').value.trim(),
      custEmail:   document.getElementById('custEmail').value.trim(),
      custCompany: document.getElementById('custCompany').value.trim(),
      invDate:     document.getElementById('invDate').value,
      dueDate:     document.getElementById('dueDate').value,
      reminderDays:parseInt(document.getElementById('reminderDays').value),
      notes:       document.getElementById('notes').value.trim(),
      items,
      subtotal: document.getElementById('subtotal').textContent,
      taxAmt:   document.getElementById('taxAmt').textContent,
      total:    document.getElementById('grandTotal').textContent,
      status:   'pending',
      sentAt:   null
    };
  }

  function validateForm(inv) {
    if (!inv.fromCompany) return 'Please enter your company name.';
    if (!inv.custName)    return 'Please enter customer name.';
    if (!inv.custEmail)   return 'Please enter customer email.';
    if (!inv.dueDate)     return 'Please set a due date.';
    if (!inv.items.length) return 'Please add at least one line item.';
    return null;
  }

  // ─── PREVIEW ──────────────────────────────────────────────────────────────────
  function previewInvoice() {
    const inv = collectForm();
    const err = validateForm(inv);
    if (err) { toast(err, true); return; }

    document.getElementById('prevFromCompany').textContent = inv.fromCompany;
    document.getElementById('prevInvNum').textContent  = inv.invNumber;
    document.getElementById('prevInvDate').textContent = 'Date: ' + fmtDate(inv.invDate);
    document.getElementById('prevCustName').textContent    = inv.custName;
    document.getElementById('prevCustCompany').textContent = inv.custCompany;
    document.getElementById('prevCustEmail').textContent   = inv.custEmail;
    document.getElementById('prevDueDate').textContent     = fmtDate(inv.dueDate);

    const today = new Date(); today.setHours(0,0,0,0);
    const due   = new Date(inv.dueDate);
    const diff  = Math.ceil((due - today) / 86400000);
    let badge = diff < 0
      ? `<span class="status-badge badge-overdue">Overdue ${Math.abs(diff)}d</span>`
      : diff === 0
      ? `<span class="status-badge badge-overdue">Due Today</span>`
      : `<span class="status-badge badge-pending">Due in ${diff} days</span>`;

    p.style.display = 'block';
    p.scrollIntoView({ behavior: 'smooth' });
  }

  // ─── SAVE ─────────────────────────────────────────────────────────────────────
  function saveInvoice() {
    const inv = collectForm();
    const err = validateForm(inv);
    if (err) { toast(err, true); return; }
    invoices.push(inv);
    localStorage.setItem('invoices', JSON.stringify(invoices));
    toast('Invoice saved successfully!');
    checkDeadlines();
  }

  // ─── SEND ─────────────────────────────────────────────────────────────────────
  async function sendNow(invData) {
    if (!config.publicKey || !config.serviceId || !config.templateId) {
      toast('Please configure EmailJS in the Setup tab first.', true);
      switchTabByName('setup');
      return;
    }

    const inv = invData || collectForm();
    if (!invData) {
      const err = validateForm(inv);
      if (err) { toast(err, true); return; }
    }

    const itemsSummary = inv.items.map(i => `${i.desc} (${i.qty} x ${fmtMoney(i.price)}) = ${fmtMoney(i.amt)}`).join('\n');

    const params = {
      to_name:        inv.custName,
      to_email:       inv.custEmail,
      from_name:      inv.fromCompany,
      from_email:     config.fromEmail || '',
      invoice_number: inv.invNumber,
      due_date:       fmtDate(inv.dueDate),
      total_amount:   fmtMoney(inv.total),
      notes:          inv.notes || 'Thank you for your business!',
      items_summary:  itemsSummary
    };

    toast('Sending email...');
    try {
      await emailjs.send(config.serviceId, config.templateId, params);
      inv.status = 'sent';
      inv.sentAt = new Date().toISOString();
      const idx = invoices.findIndex(i => i.id === inv.id);
      if (idx > -1) { invoices[idx] = inv; localStorage.setItem('invoices', JSON.stringify(invoices)); }
      toast(`Invoice sent to ${inv.custEmail}!`);
      renderList();
    } catch(e) {
      toast('Failed to send: ' + (e.text || e.message || 'Unknown error'), true);
    }
  }

  function switchTabByName(name) {
    document.querySelectorAll('.tab-btn').forEach((b,i) => {
      const names = ['create','list','setup'];
      if (names[i] === name) { b.classList.add('active'); b.click(); }
      else b.classList.remove('active');
    });
    document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
    document.getElementById('tab-' + name).classList.add('active');
  }

  // ─── LIST ─────────────────────────────────────────────────────────────────────
  function renderList() {
    const tbody = document.getElementById('invListBody');
    if (!invoices.length) {
      tbody.innerHTML = '<tr><td colspan="7" style="text-align:center;color:#aaa;padding:24px;">No invoices yet.</td></tr>';
      return;
    }
    const today = new Date(); today.setHours(0,0,0,0);
    tbody.innerHTML = invoices.slice().reverse().map(inv => {
      const due  = new Date(inv.dueDate);
      const diff = Math.ceil((due - today) / 86400000);
      let badge = inv.status === 'sent'
        ? `<span class="status-badge badge-sent">Sent</span>`
        : diff < 0
        ? `<span class="status-badge badge-overdue">Overdue</span>`
        : `<span class="status-badge badge-pending">Pending</span>`;
      return `<tr>
        <td>${inv.invNumber}</td>
        <td>${inv.custName}</td>
        <td>${inv.custEmail}</td>
        <td><strong>${fmtMoney(inv.total)}</strong></td>
        <td>${fmtDate(inv.dueDate)}</td>
        <td>${badge}</td>
        <td>
          <button class="action-btn send" onclick='sendNow(${JSON.stringify(inv)})'>&#9993; Send</button>
          <button class="action-btn del" onclick="deleteInv(${inv.id})">&#128465;</button>
        </td>
      </tr>`;
    }).join('');
  }

  function deleteInv(id) {
    if (!confirm('Delete this invoice?')) return;
    invoices = invoices.filter(i => i.id !== id);
    localStorage.setItem('invoices', JSON.stringify(invoices));
    renderList();
    checkDeadlines();
  }

  // ─── DEADLINES ────────────────────────────────────────────────────────────────
  function checkDeadlines() {
    const today = new Date(); today.setHours(0,0,0,0);
    const alerts = invoices.filter(inv => {
      if (inv.status === 'sent') return false;
      const due  = new Date(inv.dueDate);
      const diff = Math.ceil((due - today) / 86400000);
      return diff <= inv.reminderDays;
    });

    const box = document.getElementById('alertBox');
    if (!alerts.length) { box.style.display = 'none'; return; }

    box.style.display = 'block';
    document.getElementById('alertList').innerHTML = alerts.map(inv => {
      const due  = new Date(inv.dueDate);
      const diff = Math.ceil((due - today) / 86400000);
      const label = diff < 0 ? `Overdue ${Math.abs(diff)} day(s)` : diff === 0 ? 'Due TODAY' : `Due in ${diff} day(s)`;
      return `<div class="alert-item">
        <span><strong>${inv.invNumber}</strong> — ${inv.custName} — ${fmtMoney(inv.total)} — <em>${label}</em></span>
        <button onclick='sendNow(${JSON.stringify(inv)})'>&#9993; Send Now</button>
      </div>`;
    }).join('');
  }

  // ─── HELPERS ──────────────────────────────────────────────────────────────────
  function fmtDate(d) {
    if (!d) return '';
    const dt = new Date(d + 'T00:00:00');
    return dt.toLocaleDateString('en-PH', { year:'numeric', month:'long', day:'numeric' });
  }

  function fmtMoney(v) {
    return '₱' + parseFloat(v || 0).toLocaleString('en-PH', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
  }

  function toast(msg, isError = false) {
    const t = document.getElementById('toast');
    t.textContent = msg;
    t.className = isError ? 'error' : '';
    t.style.display = 'block';
    setTimeout(() => t.style.display = 'none', 3500);
  }
