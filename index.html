// ─── MSAL CONFIGURATION ─────────────────────────────────────────────────────── // Ms. Cris-an: Replace CLIENT_ID and TENANT_ID with values from Azure App Registration const MSAL_CONFIG = { auth: { clientId: 'YOUR_CLIENT_ID_HERE', authority: 'https://login.microsoftonline.com/YOUR_TENANT_ID_HERE', redirectUri: window.location.origin + window.location.pathname }, cache: { cacheLocation: 'sessionStorage', storeAuthStateInCookie: true // required for mobile Edge/Safari } }; const SCOPES = ['Files.ReadWrite', 'Files.ReadWrite.All', 'User.Read']; let _msal = null; async function initMSAL() { _msal = new msal.PublicClientApplication(MSAL_CONFIG); await _msal.initialize(); // Handle redirect response on mobile after page reload const redirectResult = await _msal.handleRedirectPromise(); if (redirectResult) { _msal.setActiveAccount(redirectResult.account); return redirectResult.account; } const accounts = _msal.getAllAccounts(); if (accounts.length > 0) { _msal.setActiveAccount(accounts[0]); return accounts[0]; } return null; } async function msalLogin() { const isMobile = /Android|iPhone|iPad|Mobile/i.test(navigator.userAgent); try { if (isMobile) { // Redirect flow — page will reload after auth await _msal.loginRedirect({ scopes: SCOPES }); return null; } const resp = await _msal.loginPopup({ scopes: SCOPES }); _msal.setActiveAccount(resp.account); return resp.account; } catch (e) { // Popup blocked — fall back to redirect if (e.errorCode === 'popup_window_error') { await _msal.loginRedirect({ scopes: SCOPES }); return null; } throw e; } } async function msalLogout() { const account = _msal?.getActiveAccount(); if (!account) return; await _msal.logoutPopup({ account }).catch(() => _msal.logoutRedirect({ account })); } async function getAccessToken() { const account = _msal?.getActiveAccount(); if (!account) throw new Error('User not logged in.'); try { const resp = await _msal.acquireTokenSilent({ scopes: SCOPES, account }); return resp.accessToken; } catch { const isMobile = /Android|iPhone|iPad|Mobile/i.test(navigator.userAgent); if (isMobile) { await _msal.acquireTokenRedirect({ scopes: SCOPES, account }); return null; } const resp = await _msal.acquireTokenPopup({ scopes: SCOPES, account }); return resp.accessToken; } } function getCurrentUser() { return _msal?.getActiveAccount() ?? null; }
